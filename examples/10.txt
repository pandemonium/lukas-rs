factorial :: Int -> Int :=
  lambda n.
    deconstruct n into
      0 -> 1
    | x ->
        let arg = x - 1
        in
            x * factorial arg

fibonacci :: Int -> Int := lambda n.
  deconstruct
    n
  into
    0 -> 1
  | 1 -> 1
  | x -> fibonacci (n - 1) + fibonacci (n - 2)

ifonacci :: Int -> Int := lambda n.
  if n = 0
  then 1
  else
    if n = 1
    then 1
    else fibonacci (n - 1) + fibonacci (n - 2)
